# 🔑 핵심 설계 원칙

- **Supabase PostgreSQL** 기반, 모든 테이블에 RLS(Row Level Security) 적용
- **KST(UTC+9)** 기준으로 날짜 계산. 서버 타임존 `Asia/Seoul` 설정 필수
- **소프트 삭제** 미사용 — 부모가 자녀 삭제 시 CASCADE DELETE
- **반올림 규칙**: 모든 XP 계산 결과는 Math.floor 적용
- **UUID v4** 를 모든 PK로 사용
- **정답은 `private` 스키마**에만 저장 — RLS + 스키마 분리 이중 보호
- **오프라인 재전송** 대응: `idempotency_key` 활용

## 3. pets (보유 펫) + equipped_pets (장착 펫)

> ⚠️ **v1.0 → v1.1 변경**:
> 

> - `pets.mood` 컬럼 추가 (스케줄러 SAD 업데이트 대상)
> 

> - `child_profiles.equipped_pet_id` 제거 → `equipped_pets` 1:1 테이블 신설 (순환 FK 제거 + 소유권 강제)
> 

```sql
CREATE TABLE pets (
  id               UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  child_id         UUID NOT NULL REFERENCES child_profiles(id) ON DELETE CASCADE,
  pet_type         TEXT NOT NULL,
  grade            pet_grade_enum NOT NULL,       -- NORMAL|RARE|EPIC|LEGEND
  xp               INT NOT NULL DEFAULT 0 CHECK (xp >= 0),
  stage            pet_stage_enum NOT NULL DEFAULT 'EGG',  -- EGG|GROWTH|AIMONG
  mood             pet_mood_enum NOT NULL DEFAULT 'IDLE',  -- HAPPY|IDLE|SAD_LIGHT|SAD_DEEP ← v1.1 신규 (기능명세서 5-4와 동기화)
  crown_unlocked   BOOLEAN NOT NULL DEFAULT FALSE,
  crown_type       crown_type_enum,
  obtained_at      TIMESTAMPTZ NOT NULL DEFAULT now(),

  UNIQUE (id, child_id),
  UNIQUE (child_id, pet_type),
  CHECK (
    (crown_unlocked = FALSE AND crown_type IS NULL)
    OR
    (crown_unlocked = TRUE AND crown_type IS NOT NULL)
  )
);

-- 장착 펫: 자녀당 1행, 자기 소유 펫만 장착 가능 (소유권 FK)
CREATE TABLE equipped_pets (
  child_id    UUID PRIMARY KEY REFERENCES child_profiles(id) ON DELETE CASCADE,
  pet_id      UUID NOT NULL UNIQUE,
  equipped_at TIMESTAMPTZ NOT NULL DEFAULT now(),

  FOREIGN KEY (pet_id, child_id) REFERENCES pets(id, child_id) ON DELETE CASCADE
);
```

| 컬럼 | 타입 | 설명 |
| --- | --- | --- |
| id | UUID PK | 펫 인스턴스 식별자 |
| child_id | UUID FK | 소유 자녀 |
| pet_type | TEXT | 펫 종류 코드 |
| grade | ENUM | NORMAL / RARE / EPIC / LEGEND |
| xp | INT | 펫 개별 XP |
| stage | ENUM | EGG(0~79) / GROWTH(80~249) / AIMONG(250+) |
| mood | ENUM | HAPPY / IDLE / SAD_LIGHT / SAD_DEEP |
| crown_unlocked | BOOLEAN | 아이몽 달성 왕관 영구 해금 |
| crown_type | ENUM | silver / gold / jewel / shining |

## 4. 미션 / 문제 / 정답 / 오늘의 AI 미션

> ⚠️ **v1.0 → v1.1 변경**: `questions` 단일 테이블 → 3계층으로 분리
> 

> - 문제 본문/보기: `question_bank` (public, RLS 적용)
> 

> - 정답/해설: `private.question_answer_keys` (private 스키마, 클라이언트 접근 불가)
> 

> - 오늘의 AI 미션: `daily_missions` + `daily_mission_questions` (v1.0에 없던 테이블)
> 

```sql
CREATE TABLE missions (
  id               UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  stage            SMALLINT NOT NULL CHECK (stage IN (1, 2, 3)),
  title            TEXT NOT NULL,
  description      TEXT,
  unlock_condition TEXT,
  is_active        BOOLEAN NOT NULL DEFAULT TRUE
);

-- 공개: 문제 본문 + 보기만
CREATE TABLE question_bank (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  mission_id    UUID REFERENCES missions(id) ON DELETE CASCADE,
  question_type question_type_enum NOT NULL,   -- OX|MULTIPLE|FILL|SITUATION
  prompt        TEXT NOT NULL,
  options       JSONB,
  source_type   question_source_enum NOT NULL DEFAULT 'STATIC',  -- STATIC|GPT|FALLBACK
  created_at    TIMESTAMPTZ NOT NULL DEFAULT now(),
  is_active     BOOLEAN NOT NULL DEFAULT TRUE
);

-- 비공개: 정답 + 해설 (private 스키마)
CREATE TABLE private.question_answer_keys (
  question_id     UUID PRIMARY KEY REFERENCES public.question_bank(id) ON DELETE CASCADE,
  answer_payload  JSONB NOT NULL,
  explanation     TEXT NOT NULL,
  created_at      TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- 클라이언트용 안전 뷰 (answer/explanation 제외)
CREATE VIEW public.question_bank_safe AS
  SELECT id, mission_id, question_type, prompt, options, source_type, created_at
  FROM public.question_bank
  WHERE is_active = TRUE;

-- 오늘의 AI 미션 헤더
CREATE TABLE daily_missions (
  id           UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  mission_date DATE NOT NULL UNIQUE,
  generated_by question_source_enum NOT NULL,
  status       daily_mission_status_enum NOT NULL DEFAULT 'GENERATED',
  created_at   TIMESTAMPTZ NOT NULL DEFAULT now(),
  archived_at  TIMESTAMPTZ,
  CHECK (
    (status = 'GENERATED' AND archived_at IS NULL)
    OR (status = 'ARCHIVED' AND archived_at IS NOT NULL)
  )
);

-- 오늘의 AI 미션 배정 문제 (최대 10개)
CREATE TABLE daily_mission_questions (
  daily_mission_id UUID NOT NULL REFERENCES daily_missions(id) ON DELETE CASCADE,
  question_id      UUID NOT NULL REFERENCES question_bank(id) ON DELETE RESTRICT,
  order_no         SMALLINT NOT NULL CHECK (order_no BETWEEN 1 AND 10),
  PRIMARY KEY (daily_mission_id, order_no),
  UNIQUE (daily_mission_id, question_id)
);
```

## 5. 미션 진행 기록 (복습 모드 지원)

> ⚠️ **v1.0 → v1.1 변경**: `mission_progress` 단일 테이블 제거 → 2개 테이블로 분리
> 

> - **문제**: `UNIQUE(child_id, mission_id, completed_at)`가 같은 날 2번 제출을 DB 레벨에서 막아 복습 모드 구현 불가
> 

> - **해결**: 시도 로그(`mission_attempts`) + 일별 요약(`mission_daily_progress`) 분리
> 

```sql
-- 제출 1건당 1행 (복습 포함)
CREATE TABLE mission_attempts (
  id               UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  child_id         UUID NOT NULL REFERENCES child_profiles(id) ON DELETE CASCADE,
  mission_id       UUID NOT NULL REFERENCES missions(id) ON DELETE CASCADE,
  attempt_date     DATE NOT NULL,
  attempt_no       INT NOT NULL CHECK (attempt_no >= 1),   -- 당일 몇 번째 시도
  score            INT NOT NULL CHECK (score >= 0),
  total            INT NOT NULL CHECK (total > 0),
  xp_earned        INT NOT NULL CHECK (xp_earned >= 0),
  is_review        BOOLEAN GENERATED ALWAYS AS (attempt_no > 1) STORED,
  idempotency_key  TEXT UNIQUE,                            -- 오프라인 재전송 중복 방지
  submitted_at     TIMESTAMPTZ NOT NULL DEFAULT now(),

  UNIQUE (child_id, mission_id, attempt_date, attempt_no),
  CHECK (score <= total)
);

CREATE INDEX idx_mission_attempts_child_date ON mission_attempts(child_id, attempt_date);

-- 하루 기준 요약 (API 조회 최적화용)
CREATE TABLE mission_daily_progress (
  child_id             UUID NOT NULL REFERENCES child_profiles(id) ON DELETE CASCADE,
  mission_id           UUID NOT NULL REFERENCES missions(id) ON DELETE CASCADE,
  progress_date        DATE NOT NULL,
  first_attempt_at     TIMESTAMPTZ NOT NULL,
  best_score           INT NOT NULL CHECK (best_score >= 0),
  total                INT NOT NULL CHECK (total > 0),
  first_xp_earned      INT NOT NULL CHECK (first_xp_earned >= 0),
  review_attempt_count INT NOT NULL DEFAULT 0 CHECK (review_attempt_count >= 0),

  PRIMARY KEY (child_id, mission_id, progress_date),
  CHECK (best_score <= total)
);
```

| 테이블 | 역할 |
| --- | --- |
| mission_attempts | 제출 이벤트 전체 로그. 복습 시도 포함 |
| mission_daily_progress | 당일 최고점 · 첫 XP · 복습 횟수 요약. API 응답용 |

## 7. 스트릭 / 공동 스트릭 / 마일스톤

> ⚠️ **v1.0 → v1.1 변경**: `friend_streaks` 구조 변경
> 

> - **문제**: `(child_a_id, child_b_id)` 구조에서는 A가 여러 명과 연결 가능 — MVP "1인 1파트너" 정책 위반
> 

> - **해결**: `child_id PK` + `partner_child_id UNIQUE` 대칭 2행 모델 (A→B, B→A 한 트랜잭션으로 insert)
> 

```sql
CREATE TABLE streak_records (
  child_id              UUID PRIMARY KEY REFERENCES child_profiles(id) ON DELETE CASCADE,
  continuous_days       INT NOT NULL DEFAULT 0 CHECK (continuous_days >= 0),
  last_completed_date   DATE,
  today_mission_count   INT NOT NULL DEFAULT 0 CHECK (today_mission_count >= 0),
  shield_count          INT NOT NULL DEFAULT 0 CHECK (shield_count >= 0),
  updated_at            TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- MVP: 1인 1파트너. A→B, B→A 대칭 2행을 한 트랜잭션에서 동시 insert
CREATE TABLE friend_streaks (
  child_id         UUID PRIMARY KEY REFERENCES child_profiles(id) ON DELETE CASCADE,
  partner_child_id UUID NOT NULL UNIQUE REFERENCES child_profiles(id) ON DELETE CASCADE,
  connected_at     TIMESTAMPTZ NOT NULL DEFAULT now(),

  CHECK (child_id <> partner_child_id)
);

CREATE TABLE milestone_rewards (
  id             UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  child_id       UUID NOT NULL REFERENCES child_profiles(id) ON DELETE CASCADE,
  milestone_days SMALLINT NOT NULL CHECK (milestone_days IN (7, 30)),
  rewarded_at    TIMESTAMPTZ NOT NULL DEFAULT now(),

  UNIQUE (child_id, milestone_days)
);

-- 사용자 목표 마일스톤 (30일 이후, 25일 단위 — 기능명세서 섹션 10-2)
CREATE TABLE streak_milestones (
  id             UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  child_id       UUID NOT NULL REFERENCES child_profiles(id) ON DELETE CASCADE,
  target_days    SMALLINT NOT NULL CHECK (target_days > 30),   -- 30일 초과만 설정 가능
  tier           SMALLINT NOT NULL CHECK (tier IN (1, 2, 3)),  -- 1단계/2단계/3단계
  achieved       BOOLEAN NOT NULL DEFAULT FALSE,
  reward_claimed BOOLEAN NOT NULL DEFAULT FALSE,
  achieved_at    TIMESTAMPTZ,
  created_at     TIMESTAMPTZ NOT NULL DEFAULT now(),

  UNIQUE (child_id, target_days),
  CHECK (reward_claimed = FALSE OR achieved = TRUE),
  CHECK ((achieved = FALSE AND achieved_at IS NULL) OR (achieved = TRUE AND achieved_at IS NOT NULL))
);

CREATE INDEX idx_streak_milestones_child ON streak_milestones(child_id);
```

**friend_streaks 연결 트랜잭션 규칙**

- A가 B와 연결 시: `(A→B)`, `(B→A)` 두 행을 **하나의 트랜잭션**에서 동시에 insert
- `child_id PK` → A는 파트너 1명만 보유 가능 (DB 레벨 보장)
- `partner_child_id UNIQUE` → B도 동시에 다른 파트너를 가질 수 없음
- 파트너 탈퇴 시 → ON DELETE CASCADE로 두 행 자동 삭제
